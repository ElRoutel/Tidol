// ===================== RENDER INTERNET ARCHIVE =====================
  async function buscarInternetArchive(query) {
  try {
    const apiUrl = `https://archive.org/advancedsearch.php?q=${encodeURIComponent(query)}+AND+mediatype:audio&fl[]=identifier,title,creator,format&sort[]=downloads+desc&rows=15&page=1&output=json`;
    const res = await fetch(apiUrl);
    const data = await res.json();

    if (!data.response?.docs?.length) return;

    const h = document.createElement('h3');
    h.textContent = 'Resultados de Internet Archive';
    resultsContainer.appendChild(h);

    data.response.docs.forEach(item => {
      const div = document.createElement('div');
      div.classList.add('search-card');

      // Detectar tipo de archivo
      const ext = item.format ? item.format.join(', ') : 'Desconocido';
      const enlace = `https://archive.org/details/${item.identifier}`;

      div.innerHTML = `
        <img src="https://archive.org/services/img/${item.identifier}" alt="cover" class="album-cover">
        <div class="card-info">
          <span class="titulo">${item.title || 'Sin t√≠tulo'}</span>
          <span class="autor">${item.creator || 'Autor desconocido'}</span>
          <span class="tipo">${ext}</span>
        </div>
      `;
      div.onclick = () => window.open(enlace, '_blank');
      resultsContainer.appendChild(div);
    });

  } catch (err) {
    console.warn('Error al buscar en Internet Archive:', err);
  }
}
